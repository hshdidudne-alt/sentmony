প্রথমবার সেটআপ

একবারই করতে হবে। নিচের SQL কোডটি Supabase-এ রান করলেই অ্যাপ সম্পূর্ণ রেডি হয়ে যাবে।

📋 মাত্র ৩টি ধাপ:
১
supabase.com/dashboard খুলুন → আপনার project → SQL Editor
২
নিচের 📋 কপি বাটন চাপুন → SQL Editor-এ পেস্ট করুন → Run চাপুন
৩
সফল হলে নিচের ✅ হয়ে গেছে বাটন চাপুন
SQL 
-- SentMoney Database Setup
-- ⚡ এই ২ লাইন প্রথমে রান করুন (Auto-Setup চালু হবে):
CREATE OR REPLACE FUNCTION exec_sql(sql text) RETURNS void LANGUAGE plpgsql SECURITY DEFINER AS $f$ BEGIN EXECUTE sql; END; $f$;
GRANT EXECUTE ON FUNCTION exec_sql(text) TO anon;
-- তারপর বাকি SQL:
CREATE TABLE IF NOT EXISTS app_config (
  id INT PRIMARY KEY DEFAULT 1,
  settings JSONB DEFAULT '{}',
  commissions JSONB DEFAULT '{}',
  vip_plans JSONB DEFAULT '[]',
  videos JSONB DEFAULT '[]',
  tasks JSONB DEFAULT '[]',
  notices JSONB DEFAULT '[]',
  admin_credentials JSONB DEFAULT '{}',
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE IF NOT EXISTS users (
  id BIGINT PRIMARY KEY,
  name TEXT NOT NULL,
  phone TEXT UNIQUE NOT NULL,
  password TEXT NOT NULL,
  vip INT DEFAULT 1,
  active BOOLEAN DEFAULT FALSE,
  balance JSONB DEFAULT '{"video":0,"task":0,"refer":0,"bonus":0}',
  refer_code TEXT UNIQUE,
  referred_by TEXT,
  team JSONB DEFAULT '[]',
  join_date TEXT,
  tasks_done_today INT DEFAULT 0,
  videos_done_today INT DEFAULT 0,
  withdraw_history JSONB DEFAULT '[]',
  earn_history JSONB DEFAULT '[]',
  created_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE IF NOT EXISTS withdrawals (
  id BIGINT PRIMARY KEY,
  user_id BIGINT,
  user_name TEXT,
  amount NUMERIC,
  charge NUMERIC,
  net NUMERIC,
  method TEXT,
  phone TEXT,
  status TEXT DEFAULT 'pending',
  date TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE IF NOT EXISTS activation_requests (
  id BIGINT PRIMARY KEY,
  user_id BIGINT,
  user_name TEXT,
  phone TEXT,
  txid TEXT,
  amount NUMERIC,
  status TEXT DEFAULT 'pending',
  date TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE IF NOT EXISTS vip_requests (
  id BIGINT PRIMARY KEY,
  user_id BIGINT,
  user_name TEXT,
  phone TEXT,
  level INT,
  plan_name TEXT,
  price NUMERIC,
  txid TEXT,
  status TEXT DEFAULT 'pending',
  date TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
ALTER TABLE app_config ENABLE ROW LEVEL SECURITY;
ALTER TABLE users ENABLE ROW LEVEL SECURITY;
ALTER TABLE withdrawals ENABLE ROW LEVEL SECURITY;
ALTER TABLE activation_requests ENABLE ROW LEVEL SECURITY;
ALTER TABLE vip_requests ENABLE ROW LEVEL SECURITY;
CREATE POLICY IF NOT EXISTS "allow_all_app_config" ON app_config FOR ALL TO anon USING (true) WITH CHECK (true);
CREATE POLICY IF NOT EXISTS "allow_all_users" ON users FOR ALL TO anon USING (true) WITH CHECK (true);
CREATE POLICY IF NOT EXISTS "allow_all_withdrawals" ON withdrawals FOR ALL TO anon USING (true) WITH CHECK (true);
CREATE POLICY IF NOT EXISTS "allow_all_activation_requests" ON activation_requests FOR ALL TO anon USING (true) WITH CHECK (true);
CREATE POLICY IF NOT EXISTS "allow_all_vip_requests" ON vip_requests FOR ALL TO anon USING (true) WITH CHECK (true);
এই সেটআপ মাত্র একবারই করতে হবে।
